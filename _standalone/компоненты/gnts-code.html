import numpy as np
import matplotlib.pyplot as plt

def bonding_curve_price(supply, a, b, c):
    """Цена минтинга токена по формуле P = A + B * T^C"""
    return a + b * np.power(supply, c)

def total_cost(supply, a, b, c, n_steps=1000):
    """Совокупная стоимость минтинга (численное интегрирование)"""
    t = np.linspace(0, supply, n_steps)
    prices = bonding_curve_price(t, a, b, c)
    return np.trapz(prices, t)

# Параметры из домашнего задания МФТИ
A, B, C = 0.10, 0.00001, 1.5
supply_range = np.linspace(0, 50000, 500)
prices = bonding_curve_price(supply_range, A, B, C)

plt.figure(figsize=(10, 5))
plt.plot(supply_range, prices, color='#7c3aed', linewidth=2)
plt.fill_between(supply_range, prices, alpha=0.1, color='#7c3aed')
plt.xlabel('Supply (токены)')
plt.ylabel('Цена минтинга ($)')
plt.title('Bonding Curve: P(T) = A + B * T^C')
plt.grid(True, alpha=0.3)
plt.show()
